Node.js / Discord Bot 
const axios = require('axios');

async function pullCard() {
  const { data } = await axios.get('https://shoob-database-api.vercel.app/api/card', {
    params: { random: true, key: 'SILENT_TECH_2026' }
  });
  
  console.log(`Pulled: ${data.name}`);
  console.log(`Image: ${data.image}`); // Auto-proxied URL
}

import requests

def pull_card():
    res = requests.get("https://shoob-database-api.vercel.app/api/card", 
        params={"random": "true", "key": "SILENT_TECH_2026"}
    )
